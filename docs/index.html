Integrate

Add to your app in 3 lines

Works with Express, Next.js, or any Node.js server. Install and wrap your routes.

$ npm install @x402charity/express 
import express from 'express';
import { x402charity } from '@x402charity/express';

const app = express();

app.use(x402charity({
  privateKey: process.env.DONATION_PRIVATE_KEY,
  charityWallet: '0x...',  // charity's wallet address
}));

// Every request now triggers a $0.001 USDC donation
app.get('/', (req, res) => res.send('Hello!'));
// middleware.ts
import { x402charity } from '@x402charity/next';

export const middleware = x402charity({
  privateKey: process.env.DONATION_PRIVATE_KEY,
  charityWallet: '0x...',
});

export const config = {
  matcher: ['/api/:path*'],
};
// Run a standalone donation server
import { startCharityServer } from '@x402charity/server';

startCharityServer({
  privateKey: process.env.DONATION_PRIVATE_KEY,
  network: 'base-sepolia',
});

// Endpoints:
// GET  /charities     — list charities
// POST /donate        — trigger donation
// GET  /dashboard     — public donation dashboard
